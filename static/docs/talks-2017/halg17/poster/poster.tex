%%
%% THIS DOES NOT WORK WITH PDFLATEX!
%%
%% \posterbox{frc}{bgc}{content}
%%
%%   puts "content" into a box with rounded corners with frame color
%%   "frc" and background color "bgc".
%%
%%   The dimensions of the box are determined by the content.
%%
%%   To get sharp corners instead of rounded corners, replace both
%%   occurrences of "linearc=2mm" by "linearc=0mm".
%%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\long\def\posterbox#1#2#3{%
  \psframebox[linecolor=#1,boxsep=true,framesep=3mm,cornersize=absolute,linearc=2mm]%
  {\psframebox*[fillcolor=#2,boxsep=false,framesep=3mm,cornersize=absolute,linearc=2mm]%
  {#3}}}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%
%% THIS DOES NOT WORK WITH PDFLATEX!
%%
%% \posterboxhw{frc}{bgc}{h}{w}{content}
%%
%%   puts "content" into a box with rounded corners with frame color
%%   "frc", background color "bgc", height "h", and width "w".
%%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\long\def\posterboxhw#1#2#3#4#5{%
  % parameters for minipage environment:
  %   top-aligned, height-2*framesep-2*linewidth,
  %   content top-aligned, width-2*framesep-2*linewidth
  %   If framesep or linewidth are changed, change here as well!
  \posterbox{#1}{#2}{\begin{minipage}[t][#3-6mm-1.6pt][t]{#4-6mm-1.6pt}
  #5
  \end{minipage}}}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%%
%% THIS DOES NOT WORK WITH PDFLATEX!
%%
%% THIS DOES NOT WORK (NICELY) WITH ROUNDED CORNERS!
%% If you want to use any of these commands, replace both occurrences of
%% "linearc=2mm" in the definition of \posterbox above by "linearc=0mm"
%% (sharp corners).
%%
%% \csection{bgc}{w}{heading}
%%
%%   makes a white section heading in a stripe colored with "bgc";
%%   "w" should be the same width as for the surrounding \posterboxhw.
%%
%% \bigcsection{bgc}{w}{heading}
%%
%%   like \csection, but the stripe is larger, touching the surrounding
%%   frame; "bgc" should preferably be the same color as the frame color
%%   "frc" of the surrounding \posterboxhw.
%%
%% \csubsection{bgc}{w}{heading}
%% \bigcsubsection{bgc}{w}{heading}
%%
%%   like \csection and \bigcsection, but using a smaller font size.
%%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\newcommand{\csection}[3]{
  \addvspace{12mm}
  \psframebox*[fillcolor=#1,boxsep=false,framesep=1.1mm,cornersize=absolute,linearc=0mm]{%
  \begin{minipage}[t][5mm][t]{#2-6mm-1.6pt}\centering\section*{\fontsize{13}{15}\selectfont\vphantom{\"(}\white #3}
  \end{minipage}}
  \vspace{2mm}
  \par}

\newcommand{\csubsection}[3]{
  \addvspace{8mm}
  \psframebox*[fillcolor=#1,boxsep=false,framesep=1.1mm,cornersize=absolute,linearc=0mm]{%
  \begin{minipage}[t][5mm][t]{#2-6mm-1.6pt}\centering\subsection*{\vphantom{\"(}\white #3}
  \end{minipage}}
  \vspace{1mm}
  \par}

\newcommand{\bigcsection}[3]{
  \addvspace{12mm}
  \psframebox*[fillcolor=#1,boxsep=false,framesep=3.0mm,cornersize=absolute,linearc=0mm]{%
  \begin{minipage}[t][3mm][t]{#2-6mm-1.6pt}\subsection*{\fontsize{13}{15}\selectfont\vphantom{(}\white \smash{#3}}
  \end{minipage}}
  \vspace{1mm}
  \par}

\newcommand{\bigcsubsection}[3]{
  \addvspace{12mm}
  \psframebox*[fillcolor=#1,boxsep=false,framesep=3.0mm,cornersize=absolute,linearc=0mm]{%
  \begin{minipage}[t][2mm][t]{#2-6mm-1.6pt}\subsection*{\vphantom{(}\white \smash{#3}}
  \end{minipage}}
  \vspace{1mm}
  \par}
